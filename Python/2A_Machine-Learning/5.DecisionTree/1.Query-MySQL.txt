    SELECT 
        T0.id,
        T0.fecha,
        T0.genero,
    CASE 
        WHEN T0.genero='femenino' THEN 1 
        WHEN T0.genero='masculino' THEN 2 
        ELSE 0 END as sexo,
        T0.origen,
    CASE 
        WHEN T0.origen='udemy' THEN 1 
        WHEN T0.origen='frogames' THEN 2
        WHEN T0.origen='crehana' THEN 3
        WHEN T0.origen='facebook' THEN 4
        WHEN T0.origen='web' THEN 5
        ELSE 6 END as source,
    T0.edad,
    T1.corr as curso,
    T1.name as nombre_curso,
    T0.mensaje as comentario,
    T0.tickmarks as tick,
    CASE 
        WHEN T0.tickmarks <7 THEN 'Detractor' 
        WHEN T0.tickmarks >8 THEN 'Promotor' 
        ELSE 'Pasivo' END as Clasificacion
    FROM `tb_nps` T0 JOIN curso_nps T1 on T0.producto = T1.id
